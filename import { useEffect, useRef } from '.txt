import { useEffect, useRef } from 'react';

const services = [
  "UI/UX Design",
  "Frontend Development",
  "Backend Development",
  "Full Stack Projects",
  "Open Source Collab",
  "Internship & Work",
];

const MarqueeRow = () => {
  const setARef   = useRef(null); // first copy — measured for loop point
  const wrapRef   = useRef(null);
  const rafRef    = useRef(null);
  const offsetRef = useRef(0);
  const halfRef   = useRef(0);

  useEffect(() => {
    // wait one frame so DOM is painted and widths are real
    const raf = requestAnimationFrame(() => {
      if (!setARef.current) return;
      halfRef.current = setARef.current.scrollWidth;

      const tick = () => {
        offsetRef.current -= 0.8;
        if (Math.abs(offsetRef.current) >= halfRef.current) {
          offsetRef.current = 0;
        }
        if (wrapRef.current) {
          wrapRef.current.style.transform = `translateX(${offsetRef.current}px)`;
        }
        rafRef.current = requestAnimationFrame(tick);
      };

      tick();
    });

    return () => {
      cancelAnimationFrame(raf);
      cancelAnimationFrame(rafRef.current);
    };
  }, []);

  const pause  = () => cancelAnimationFrame(rafRef.current);
  const resume = () => {
    const tick = () => {
      offsetRef.current -= 0.8;
      if (Math.abs(offsetRef.current) >= halfRef.current) {
        offsetRef.current = 0;
      }
      if (wrapRef.current) {
        wrapRef.current.style.transform = `translateX(${offsetRef.current}px)`;
      }
      rafRef.current = requestAnimationFrame(tick);
    };
    tick();
  };

  const linkProps = {
    href: "mailto:daniel100setiawan@gmail.com?subject=Let's work together",
  };

  const ItemSet = ({ innerRef }) => (
    <span className="lc-marquee-set" ref={innerRef}>
      {services.map((label, i) => (
        <a key={i} {...linkProps}>
          {label}<span className="sep">✦</span>
        </a>
      ))}
    </span>
  );

  return (
    <div className="lc-row" onMouseEnter={pause} onMouseLeave={resume}>
      <div className="lc-row-bg" />
      <div className="lc-marquee" ref={wrapRef}>
        {/* set A — measured */}
        <ItemSet innerRef={setARef} />
        {/* set B — identical clone, sits right after A */}
        <ItemSet innerRef={null} />
      </div>
      <span className="lc-row-arrow">↗</span>
    </div>
  );
};
